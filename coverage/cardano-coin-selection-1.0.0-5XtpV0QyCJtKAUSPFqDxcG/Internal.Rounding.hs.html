    9 -- Provides internal types and functions relating to rounding of fractional

   12 module Internal.Rounding
   13     ( RoundingDirection (..)

   16 
   17 import Prelude hiding
   18     ( round )

   20 -- | Indicates a rounding direction to be used when converting from a
   21 --   fractional value to an integral value.
   22 --
   23 -- See 'round'.
   24 --
   25 data RoundingDirection
   26     = RoundUp
   27       -- ^ Round up to the nearest integral value.
   28     | RoundDown
   29       -- ^ Round down to the nearest integral value.
   30     deriving (Eq, Show)
   31 
   32 -- | Use the given rounding direction to round the given fractional value,
   33 --   producing an integral result.
   34 --
   35 round :: (RealFrac a, Integral b) => RoundingDirection -> a -> b
   36 round = \case
   37     RoundUp -> ceiling
   38     RoundDown -> floor
