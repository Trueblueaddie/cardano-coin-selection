   56 -- | Represents a non-negative integral amount of currency.
   57 --
   58 -- Use 'coinFromNatural' to create a coin from a natural number.
   59 --
   60 -- Use 'coinToNatural' to convert a coin into a natural number.
   61 --
   62 -- @since 1.0.0
   63 newtype Coin = Coin { unCoin :: Natural }
   64     deriving stock (Eq, Generic, Ord)
   65     deriving Show via (Quiet Coin)
   66 
   67 -- | Creates a coin from an integral number.
   68 --
   69 -- Returns a coin if (and only if) the given input is not negative.
   70 --
   71 coinFromIntegral :: Integral i => i -> Maybe Coin
   72 coinFromIntegral i
   73     | i >= 0    = Just $ Coin $ Prelude.fromIntegral i
   74     | otherwise = Nothing
   75 
   76 -- | Creates a coin from a natural number.
   77 --
   78 -- @since 1.0.0
   79 coinFromNatural :: Natural -> Coin
   80 coinFromNatural = Coin
   81 
   82 -- | Converts the given coin into an integral number.
   83 --
   84 coinToIntegral :: Integral i => Coin -> i
   85 coinToIntegral (Coin i) = Prelude.fromIntegral i
   86 
   87 -- | Converts the given coin into a natural number.
   88 --
   89 -- @since 1.0.0
   90 coinToNatural :: Coin -> Natural
   91 coinToNatural = unCoin
   92 
   93 add :: Coin -> Coin -> Coin
   94 add (Coin x) (Coin y) = Coin $ x + y
   95 
   96 sub :: Coin -> Coin -> Maybe Coin
   97 sub (Coin x) (Coin y) = coinFromIntegral $ toInteger x - toInteger y
   98 
   99 mul :: Integral i => Coin -> i -> Maybe Coin
  100 mul (Coin x) y = coinFromIntegral $ toInteger x * toInteger y
  101 
  102 div :: Integral i => Coin -> i -> Maybe Coin
  103 div (Coin x) y
  104     | y <= 0    = Nothing
  105     | otherwise = coinFromIntegral $ toInteger x `Prelude.div` toInteger y
  106 
  107 mod :: Integral i => Coin -> i -> Maybe Coin
  108 mod (Coin x) y
  109     | y <= 0    = Nothing
  110     | otherwise = coinFromIntegral $ toInteger x `Prelude.mod` toInteger y
  111 
  112 distance :: Coin -> Coin -> Coin
  113 distance (Coin x) (Coin y)
  114     | x >= y    = Coin $ x - y
  115     | otherwise = Coin $ y - x
  116 
  117 pred :: Coin -> Maybe Coin
  118 pred x = x `sub` one
  119 
  120 succ :: Coin -> Coin
  121 succ x = x `add` one
  122 
  123 isZero :: Coin -> Bool
  124 isZero = (== zero)
  125 
  126 zero :: Coin
  127 zero = Coin 0
  128 
  129 one :: Coin
  130 one = Coin 1
