   61 --------------------------------------------------------------------------------
   62 -- Coin Selection for Migration
   63 --------------------------------------------------------------------------------
   64 
   65 -- | Creates a __series__ of coin selections that, when published as
   66 --   transactions, will have the effect of migrating all funds from one
   67 --   wallet to another.
   68 --
   69 -- Since UTxO-based blockchains typically impose limits on the sizes of
   70 -- individual transactions, and since individual UTxO sets can contain
   71 -- /arbitrarily/ many entries, migrating all funds from one wallet to another
   72 -- may require the creation of /several/ transactions.
   73 --
   74 -- This function therefore /partitions/ the given set of inputs into multiple
   75 -- /batches/ of up to __/b/__ inputs, where __/b/__ is specified by the given
   76 -- 'BatchSize' parameter. (See 'idealBatchSize' for an automatic way to
   77 -- calculate a suitable batch size.)
   78 --
   79 -- For each batch of inputs, this function creates a separate 'CoinSelection'
   80 -- with the given 'inputs' /and/ a generated 'change' set, where the 'change'
   81 -- set represents the value to be transferred to the target wallet, carefully
   82 -- adjusted to deduct a fee in accordance with the given 'FeeOptions'
   83 -- parameter. The set of 'outputs' for each coin selection is /purposefully/
   84 -- left empty, as /all/ value is captured in the 'change' set.
   85 --
   86 -- @since 1.0.0
   87 selectCoins
   88     :: forall i o . (Ord i, Ord o)
   89     => FeeOptions i o
   90         -- ^ The fee options.
   91     -> BatchSize
   92         -- ^ The maximum number of inputs to include in each selection.
   93     -> CoinMap i
   94         -- ^ The UTxO set to migrate.
   95     -> [CoinSelection i o]
   96 selectCoins options (BatchSize batchSize) utxo =
   97     evalState migrate (coinMapToList utxo)
   98   where
   99     FeeOptions {dustThreshold, feeEstimator, feeBalancingPolicy} = options
  100 
  101     migrate :: State [CoinMapEntry i] [CoinSelection i o]
  102     migrate = do
  103         batch <- getNextBatch
  104         if null batch then
  105             pure []
  106         else case adjustForFee (mkCoinSelection batch) of
  107             Nothing -> pure []
  108             Just coinSel -> do
  109                 rest <- migrate
  110                 pure (coinSel:rest)
  111 
  112     -- Construct a provisional 'CoinSelection' from the given selected inputs.
  113     -- Note that the selection may look a bit weird at first sight as it has
  114     -- no outputs (we are paying everything to ourselves!).
  115     mkCoinSelection :: [CoinMapEntry i] -> CoinSelection i o
  116     mkCoinSelection inps = CoinSelection
  117         { inputs = coinMapFromList inps
  118         , outputs = mempty
  119         , change =
  120             let chgs = mapMaybe (noDust . entryValue) inps
  121             in if null chgs then [threshold] else chgs
  122         }
  123       where
  124         threshold = unDustThreshold dustThreshold
  125         noDust :: Coin -> Maybe Coin
  126         noDust c
  127             | c < threshold = Nothing
  128             | otherwise = Just c
  129 
  130     -- | Attempt to balance the coin selection by reducing or increasing the
  131     -- change values based on the computed fees.
  132     adjustForFee :: CoinSelection i o -> Maybe (CoinSelection i o)
  133     adjustForFee !coinSel = case change coinSel of
  134         -- If there's no change, nothing to adjust
  135         [] -> Nothing
  136 
  137         -- No difference between required and computed, we're done
  138         (_ : _) | diff == 0 -> Just coinSel
  139 
  140         -- Otherwise, we have 2 cases:
  141         --
  142         -- 1/ diff < 0
  143         -- We aren't giving enough as fee, so we need to reduce one output.
  144         --
  145         -- 2/ diff > 0
  146         -- We have some surplus so we add it to an arbitrary output
  147         --
  148         -- If both cases we can simply modify one output by adding `diff`, the
  149         -- sign of `diff` making for the right modification.
  150         -- We then recursively call ourselves for this might reduce the number
  151         -- of outputs and change the fee.
  152         (c : cs) -> do
  153             let coinSel' = coinSel
  154                     { change = modifyFirst (c :| cs) (applyDiff diff) }
  155             let costOfSurplus
  156                     = fromIntegral
  157                     $ C.coinToNatural
  158                     $ C.distance
  159                         (unFee $ estimateFee feeEstimator coinSel')
  160                         (unFee $ estimateFee feeEstimator coinSel )
  161             if
  162                 -- Adding the change costs less than not having it, so it's
  163                 -- worth trying.
  164                 | costOfSurplus < actualFee ->
  165                     adjustForFee coinSel'
  166 
  167                 -- Adding the change costs more than not having it, If we don't
  168                 -- require strict balancing, we can leave the selection as-is.
  169                 | feeBalancingPolicy == RequireMinimalFee ->
  170                     pure coinSel
  171 
  172                 -- Adding the change costs more than not having it. So,
  173                 -- depending on our balancing policy, we may stop the balancing
  174                 -- right here, or, if we must balance the selection discard the
  175                 -- whole selection: it can't be balanced with this algorithm.
  176                 --
  177                 -- Note that this last extreme case is reached when using an
  178                 -- unstable fee policy (where values of outputs can influence
  179                 -- the policy) AND, require transactions to be 100% balanced.
  180                 -- This is a silly thing to do.
  181                 | otherwise ->
  182                     Nothing
  183       where
  184         applyDiff :: Integer -> Coin -> Coin
  185         applyDiff i c
  186             = fromMaybe C.zero
  187             $ coinFromIntegral (i + coinToIntegral c)
  188 
  189         diff :: Integer
  190         diff = actualFee - requiredFee
  191           where
  192             requiredFee
  193                 = coinToIntegral $ unFee
  194                 $ estimateFee feeEstimator coinSel
  195 
  196         actualFee :: Integer
  197         actualFee
  198             = coinToIntegral (sumInputs coinSel)
  199             - coinToIntegral (sumChange coinSel)
  200 
  201     -- | Apply the given function to the first coin of the list. If the
  202     -- operation makes the 'Coin' smaller than the dust threshold, the coin is
  203     -- discarded.
  204     modifyFirst :: NonEmpty Coin -> (Coin -> Coin) -> [Coin]
  205     modifyFirst (c :| cs) op
  206         | c' <= threshold = cs
  207         | otherwise = c' : cs
  208       where
  209         c' = op c
  210         threshold = unDustThreshold dustThreshold
  211 
  212     getNextBatch :: State [a] [a]
  213     getNextBatch = do
  214         xs <- get
  215         let (batch, rest) = splitAt (fromIntegral batchSize) xs
  216         put rest
  217         pure batch

  219 -- | An upper limit for the number of 'inputs' to include in each coin selection
  220 --   generated by 'selectCoins'.
  221 --
  222 -- @since 1.0.0
  223 newtype BatchSize = BatchSize Word16
  224     deriving (Eq, Generic, Ord, Show)
  225 
  226 -- | Calculate an ideal batch size based on the given coin selection limit.
  227 --
  228 -- @since 1.0.0
  229 idealBatchSize :: CoinSelectionLimit -> BatchSize
  230 idealBatchSize coinselOpts = BatchSize $ fixPoint 1
  231   where
  232     fixPoint :: Word16 -> Word16
  233     fixPoint !n
  234         | maxN n <= n = n
  235         | n == maxBound = n
  236         | otherwise = fixPoint (n + 1)
  237       where
  238         maxN :: Word16 -> Word16
  239         maxN = calculateLimit coinselOpts
