  133 newtype FeeAdjustmentError
  134     = ErrCannotCoverFee Fee
  135     -- ^ UTxO exhausted during fee covering
  136     -- We record what amount missed to cover the fee
  137     deriving (Show, Eq)

  139 -- | Given the coin selection result from a policy run, adjust the outputs for
  140 -- fees, potentially returning additional inputs that we need to cover all
  141 -- fees.
  142 --
  143 -- We lose the relationship between the transaction outputs and their
  144 -- corresponding inputs/change outputs here. This is a decision we may wish to
  145 -- revisit later. For now however note that since
  146 --
  147 --  (a) coin selection tries to establish a particular ratio between
  148 --      payment outputs and change outputs (currently it aims for an average of
  149 --      1:1)
  150 --
  151 --  (b) coin selection currently only generates a single change output per
  152 --      payment output, distributing the fee proportionally across all change
  153 --      outputs is roughly equivalent to distributing it proportionally over
  154 --      the payment outputs (roughly, not exactly, because the 1:1 proportion
  155 --      is best effort only, and may in some cases be wildly different).
  156 --
  157 -- Note that for (a) we don't need the ratio to be 1:1, the above reasoning
  158 -- will remain true for any proportion 1:n. For (b) however, if coin selection
  159 -- starts creating multiple outputs, and this number may vary, then losing the
  160 -- connection between outputs and change outputs will mean that that some
  161 -- outputs may pay a larger percentage of the fee (depending on how many change
  162 -- outputs the algorithm happened to choose).
  163 --
  164 adjustForFee
  165     :: (Ord i, Show i, Show o, MonadRandom m)
  166     => FeeOptions i o
  167     -> CoinMap i
  168     -> CoinSelection i o
  169     -> ExceptT FeeAdjustmentError m (CoinSelection i o)
  170 adjustForFee unsafeOpt utxo coinSel = do
  171     let opt = invariant
  172             "adjustForFee: fee must be non-null" unsafeOpt (not . nullFee)
  173     senderPaysFee opt utxo coinSel
  174   where
  175     nullFee opt = estimateFee (feeEstimator opt) coinSel == Fee C.zero

  181 -- Calculates the current fee associated with a given 'CoinSelection'.
  182 --
  183 -- If the result is less than zero, returns 'Nothing'.
  184 --
  185 calculateFee :: CoinSelection i o -> Maybe Fee
  186 calculateFee s = Fee <$> sumInputs s `C.sub` (sumOutputs s `C.add` sumChange s)
  187 
  188 -- The sender pays fee in this scenario, so fees are removed from the change
  189 -- outputs, and new inputs are selected if necessary.
  190 --
  191 senderPaysFee
  192     :: forall i o m . (Ord i, Show i, Show o, MonadRandom m)
  193     => FeeOptions i o
  194     -> CoinMap i
  195     -> CoinSelection i o
  196     -> ExceptT FeeAdjustmentError m (CoinSelection i o)
  197 senderPaysFee FeeOptions {feeEstimator, dustThreshold} utxo sel =
  198     evalStateT (go sel) utxo
  199   where
  200     go
  201         :: CoinSelection i o
  202         -> StateT (CoinMap i) (ExceptT FeeAdjustmentError m) (CoinSelection i o)
  203     go coinSel@(CoinSelection inps outs chgs) = do
  204         -- 1/
  205         -- We compute fees using all inputs, outputs and changes since all of
  206         -- them have an influence on the fee calculation.
  207         let feeUpperBound = estimateFee feeEstimator coinSel
  208         -- 2/
  209         -- Substract fee from change outputs, proportionally to their value.
  210         let chgs' = reduceChangeOutputs dustThreshold feeUpperBound chgs
  211         let coinSel' = coinSel { change = chgs' }
  212         let remFee = remainingFee feeEstimator coinSel'
  213         -- 3.1/
  214         -- Should the change cover the fee, we're (almost) good. By removing
  215         -- change outputs, we make them smaller and may reduce the size of the
  216         -- transaction, and the fee. Thus, we end up paying slightly more than
  217         -- the upper bound. We could do some binary search and try to
  218         -- re-distribute excess across changes until fee becomes bigger.
  219         if remFee == Fee C.zero
  220         then pure coinSel'
  221         else do
  222             -- 3.2/
  223             -- Otherwise, we need an extra entries from the available utxo to
  224             -- cover what's left. Note that this entry may increase our change
  225             -- because we may not consume it entirely. So we will just split
  226             -- the extra change across all changes possibly increasing the
  227             -- number of change outputs (if there was none, or if increasing a
  228             -- change value causes an overflow).
  229             --
  230             -- Because selecting a new input increases the fee, we need to
  231             -- re-run the algorithm with this new elements and using the initial
  232             -- change plus the extra change brought up by this entry and see if
  233             -- we can now correctly cover fee.
  234             inps' <- coverRemainingFee remFee
  235             let extraChange = splitCoin (sumEntries inps') chgs
  236             go $ CoinSelection (inps <> coinMapFromList inps') outs extraChange
  237 
  238 -- A short and simple version of the 'random' fee policy to cover for the fee
  239 -- in the case where existing set of change is not enough.
  240 --
  241 coverRemainingFee
  242     :: MonadRandom m
  243     => Fee
  244     -> StateT (CoinMap i) (ExceptT FeeAdjustmentError m) [CoinMapEntry i]
  245 coverRemainingFee (Fee fee) = go [] where
  246     go acc
  247         | sumEntries acc >= fee =
  248             return acc
  249         | otherwise = do
  250             -- We ignore the size of the fee, and just pick randomly
  251             StateT (lift . coinMapRandomEntry) >>= \case
  252                 Just entry ->
  253                     go (entry : acc)
  254                 Nothing ->
  255                     lift $ throwE $ ErrCannotCoverFee $ Fee $
  256                         fee `C.distance` (sumEntries acc)

  258 -- Pays for the given fee by subtracting it from the given list of change
  259 -- outputs, so that each change output is reduced by a portion of the fee
  260 -- that's in proportion to its relative size.
  261 --
  262 -- == Basic Examples
  263 --
  264 -- >>> reduceChangeOutputs (DustThreshold 0) (Fee 4) (Coin <$> [2, 2, 2, 2])
  265 -- [Coin 1, Coin 1, Coin 1, Coin 1]
  266 --
  267 -- >>> reduceChangeOutputs (DustThreshold 0) (Fee 15) (Coin <$> [2, 4, 8, 16])
  268 -- [Coin 1, Coin 2, Coin 4, Coin 8]
  269 --
  270 -- == Handling Dust
  271 --
  272 -- Any dust outputs in the resulting list are coalesced according to the given
  273 -- dust threshold: (See 'coalesceDust'.)
  274 --
  275 -- >>> reduceChangeOutputs (DustThreshold 1) (Fee 4) (Coin <$> [2, 2, 2, 2])
  276 -- [Coin 4]
  277 --
  278 -- == Handling Insufficient Change
  279 --
  280 -- If there's not enough change to pay for the fee, or if there's only just
  281 -- enough to pay for it exactly, this function returns the /empty list/:
  282 --
  283 -- >>> reduceChangeOutputs (DustThreshold 0) (Fee 15) (Coin <$> [10])
  284 -- []
  285 --
  286 -- >>> reduceChangeOutputs (DustThreshold 0) (Fee 15) (Coin <$> [1, 2, 4, 8])
  287 -- []
  288 --
  289 reduceChangeOutputs :: DustThreshold -> Fee -> [Coin] -> [Coin]
  290 reduceChangeOutputs threshold (Fee totalFee) changeOutputs
  291     | totalFee >= totalChange =
  292         []
  293     | otherwise =
  294         case positiveChangeOutputs of
  295             x : xs -> x :| xs
  296                 & distributeFee (Fee totalFee)
  297                 & fmap payFee
  298                 & coalesceDust threshold
  299             [] -> []
  300   where
  301     payFee :: (Fee, Coin) -> Coin
  302     payFee (Fee f, c) =
  303         fromMaybe C.zero (c `C.sub` f)
  304 
  305     positiveChangeOutputs :: [Coin]
  306     positiveChangeOutputs = filter (> C.zero) changeOutputs
  307 
  308     totalChange = F.fold changeOutputs

  310 -- Distribute the given fee over the given list of coins, so that each coin
  311 -- is allocated a __fraction__ of the fee in proportion to its relative size.
  312 --
  313 -- == Pre-condition
  314 --
  315 -- Every coin in the given list must be __non-zero__ in value.
  316 --
  317 -- == Examples
  318 --
  319 -- >>> distributeFee (Fee 2) [(Coin 1), (Coin 1)]
  320 -- [(Fee 1, Coin 1), (Fee 1, Coin 1)]
  321 --
  322 -- >>> distributeFee (Fee 4) [(Coin 1), (Coin 1)]
  323 -- [(Fee 2, Coin 1), (Fee 2, Coin 1)]
  324 --
  325 -- >>> distributeFee (Fee 7) [(Coin 1), (Coin 2), (Coin 4)]
  326 -- [(Fee 1, Coin 1), (Fee 2, Coin 2), (Fee 4, Coin 4)]
  327 --
  328 -- >>> distributeFee (Fee 14) [(Coin 1), (Coin 2), (Coin 4)]
  329 -- [(Fee 2, Coin 1), (Fee 4, Coin 2), (Fee 8, Coin 4)]
  330 --
  331 distributeFee :: Fee -> NonEmpty Coin -> NonEmpty (Fee, Coin)
  332 distributeFee (Fee feeTotal) coinsUnsafe =
  333     NE.zip feesRounded coins
  334   where
  335     -- A list of coins that are non-zero in value.
  336     coins :: NonEmpty Coin
  337     coins =
  338         invariant "distributeFee: all coins must be non-zero in value."
  339         coinsUnsafe (C.zero `F.notElem`)
  340 
  341     -- A list of rounded fee portions, where each fee portion deviates from the
  342     -- ideal unrounded portion by as small an amount as possible.
  343     feesRounded :: NonEmpty Fee
  344     feesRounded
  345         -- 1. Start with the list of ideal unrounded fee portions for each coin:
  346         = feesUnrounded
  347         -- 2. Attach an index to each fee portion, so that we can remember the
  348         --    original order:
  349         & NE.zip indices
  350         -- 3. Sort the fees into descending order of their fractional parts:
  351         & NE.sortBy (comparing (Down . fractionalPart . snd))
  352         -- 4. Apply pre-computed roundings to each fee portion:
  353         --    * portions with the greatest fractional parts are rounded up;
  354         --    * portions with the smallest fractional parts are rounded down.
  355         & NE.zipWith (\roundDir (i, f) -> (i, round roundDir f)) feeRoundings
  356         -- 5. Restore the original order:
  357         & NE.sortBy (comparing fst)
  358         -- 6. Strip away the indices:
  359         & fmap snd
  360         -- 7. Transform results into fees:
  361         & fmap (Fee . fromMaybe C.zero . C.coinFromIntegral @Integer)
  362       where
  363         indices :: NonEmpty Int
  364         indices = 0 :| [1 ..]
  365 
  366     -- A list of rounding directions, one per fee portion.
  367     --
  368     -- Since the ideal fee portion for each coin is a rational value, we must
  369     -- therefore round each rational value either /up/ or /down/ to produce a
  370     -- final integer result.
  371     --
  372     -- However, we can't take the simple approach of either rounding /all/ fee
  373     -- portions down or rounding /all/ fee portions up, as this could cause the
  374     -- sum of fee portions to either undershoot or overshoot the original fee.
  375     --
  376     -- So in order to hit the fee exactly, we must round /some/ of the portions
  377     -- up, and /some/ of the portions down.
  378     --
  379     -- Fortunately, we can calculate exactly how many fee portions must be
  380     -- rounded up, by first rounding /all/ portions down, and then computing
  381     -- the /shortfall/ between the sum of the rounded-down portions and the
  382     -- original fee.
  383     --
  384     -- We return a list where all values of 'RoundUp' occur in a contiguous
  385     -- section at the start of the list, of the following form:
  386     --
  387     --     [RoundUp, RoundUp, ..., RoundDown, RoundDown, ...]
  388     --
  389     feeRoundings :: NonEmpty RoundingDirection
  390     feeRoundings =
  391         applyN feeShortfall (NE.cons RoundUp) (NE.repeat RoundDown)
  392       where
  393          -- The part of the total fee that we'd lose if we were to take the
  394          -- simple approach of rounding all ideal fee portions /down/.
  395         feeShortfall
  396             = C.coinToIntegral feeTotal
  397             - fromIntegral @Integer (F.sum $ round RoundDown <$> feesUnrounded)
  398 
  399     -- A list of ideal unrounded fee portions, with one fee portion per coin.
  400     --
  401     -- A coin's ideal fee portion is the rational portion of the total fee that
  402     -- corresponds to that coin's relative size when compared to other coins.
  403     feesUnrounded :: NonEmpty Rational
  404     feesUnrounded = calculateIdealFee <$> coins
  405       where
  406         calculateIdealFee c
  407             = C.coinToIntegral c
  408             * C.coinToIntegral feeTotal
  409             % C.coinToIntegral totalCoinValue
  410 
  411     -- The total value of all coins.
  412     totalCoinValue :: Coin
  413     totalCoinValue = F.fold coins

  415 -- From the given list of coins, remove dust coins with a value less than or
  416 -- equal to the given threshold value, redistributing their total value over
  417 -- the coins that remain.
  418 --
  419 -- This function satisfies the following properties:
  420 --
  421 -- >>> sum coins = sum (coalesceDust threshold coins)
  422 -- >>> all (/= Coin 0) (coalesceDust threshold coins)
  423 --
  424 coalesceDust :: DustThreshold -> NonEmpty Coin -> [Coin]
  425 coalesceDust (DustThreshold threshold) coins =
  426     splitCoin valueToDistribute coinsToKeep
  427   where
  428     (coinsToKeep, coinsToRemove) = NE.partition (> threshold) coins
  429     valueToDistribute = F.fold coinsToRemove

  431 -- Computes how much is left to pay given a particular selection.
  432 --
  433 remainingFee
  434     :: (HasCallStack, Show i, Show o)
  435     => FeeEstimator i o
  436     -> CoinSelection i o
  437     -> Fee
  438 remainingFee FeeEstimator {estimateFee} s
  439     | fee >= diff =
  440         Fee (fee `C.distance` diff)
  441     | feeDangling >= diff =
  442         Fee (feeDangling `C.distance` fee)
  443     | otherwise =
  444         -- NOTE
  445         -- The only case where we may end up with an unbalanced transaction is
  446         -- when we have a dangling change output (i.e. adding it costs too much
  447         -- and we can't afford it, but not having it result in too many coins
  448         -- left for fees).
  449         error $ unwords
  450             [ "Generated an unbalanced tx! Too much left for fees"
  451             , ": fee (raw) =", show fee
  452             , ": fee (dangling) =", show feeDangling
  453             , ", diff =", show diff
  454             , "\nselection =", show s
  455             ]
  456   where
  457     Fee fee = estimateFee s
  458     Fee diff = fromMaybe errorUnderfundedSelection (calculateFee s)
  459     Fee feeDangling = estimateFee s { change = [diff `C.distance` fee] }
  460     errorUnderfundedSelection = error
  461         "Cannot calculate remaining fee for an underfunded selection."

  463 -- Splits up the given coin of value __@v@__, distributing its value over the
  464 -- given coin list of length __@n@__, so that each coin value is increased by
  465 -- an integral amount within unity of __@v/n@__, producing a new list of coin
  466 -- values where the overall total is preserved.
  467 --
  468 -- == Basic Examples
  469 --
  470 -- When it's possible to divide a coin evenly, each coin value is increased by
  471 -- the same integer amount:
  472 --
  473 -- >>> splitCoin (Coin 40) (Coin <$> [1, 1, 1, 1])
  474 -- [Coin 11, Coin 11, Coin 11, Coin 11]
  475 --
  476 -- >>> splitCoin (Coin 40) (Coin <$> [1, 2, 3, 4])
  477 -- [Coin 11, Coin 12, Coin 13, Coin 14]
  478 --
  479 -- == Handling Non-Uniform Increases
  480 --
  481 -- When it's not possible to divide a coin evenly, each integral coin value in
  482 -- the resulting list is always within unity of the ideal unrounded result:
  483 --
  484 -- >>> splitCoin (Coin 2) (Coin <$> [1, 1, 1, 1])
  485 -- [Coin 1, Coin 1, Coin 2, Coin 2]
  486 --
  487 -- >>> splitCoin (Coin 10) (Coin <$> [1, 1, 1, 1])
  488 -- [Coin 3, Coin 3, Coin 4, Coin 4]
  489 --
  490 -- == Handling Empty Lists
  491 --
  492 -- If the given list is empty, this function returns a list with the original
  493 -- given coin as its sole element:
  494 --
  495 -- >>> splitCoin (Coin 10) []
  496 -- [Coin 10]
  497 --
  498 -- == Properties
  499 --
  500 -- The total value is always preserved:
  501 --
  502 -- >>> sum (splitCoin x ys) == x + sum ys
  503 --
  504 splitCoin :: Coin -> [Coin] -> [Coin]
  505 splitCoin coinToSplit coinsToIncrease =
  506     case (mIncrement, mShortfall) of
  507         (Just increment, Just shortfall) ->
  508             zipWith C.add coinsToIncrease increments
  509           where
  510             increments = zipWith C.add majorIncrements minorIncrements
  511             majorIncrements = repeat increment
  512             minorIncrements = replicate (C.coinToIntegral shortfall) C.one
  513                 <> repeat C.zero
  514         _ | coinToSplit > C.zero ->
  515             [coinToSplit]
  516         _ ->
  517             []
  518   where
  519     mCoinCount = length coinsToIncrease
  520     mIncrement = coinToSplit `C.div` mCoinCount
  521     mShortfall = coinToSplit `C.mod` mCoinCount
