   85 --------------------------------------------------------------------------------
   86 -- Coin Map
   87 --------------------------------------------------------------------------------
   88 
   89 -- | A mapping from unique keys to associated 'Coin' values.
   90 --
   91 -- A 'CoinMap' can be used to represent:
   92 --
   93 --   * a UTxO set, where each key within the map refers to an unspent output
   94 --     from a previous transaction.
   95 --
   96 --   * a set of 'inputs' to a 'CoinSelection', where each input is an entry
   97 --     selected from a UTxO set by a 'CoinSelectionAlgorithm'.
   98 --
   99 --   * a set of 'outputs' for a 'CoinSelection', where each key within the map
  100 --     refers to the address of a payment recipient.
  101 --
  102 -- A 'CoinMap' can be constructed with the 'coinMapFromList' function.
  103 --
  104 -- The total value of a 'CoinMap' is given by the 'coinMapValue' function.
  105 --
  106 newtype CoinMap a = CoinMap { unCoinMap :: Map a Coin }
  107     deriving (Eq, Generic)
  108     deriving Show via (Quiet (CoinMap a))
  109 
  110 instance Foldable CoinMap where
  111     foldMap f = F.fold . fmap (f . entryKey) . coinMapToList
  112 
  113 instance Ord a => Monoid (CoinMap a) where
  114     mempty = CoinMap mempty
  115 
  116 instance Ord a => Semigroup (CoinMap a) where
  117     CoinMap a <> CoinMap b = CoinMap $ Map.unionWith (<>) a b
  118 
  119 -- | An entry for a 'CoinMap'.
  120 --
  121 data CoinMapEntry a = CoinMapEntry
  122     { entryKey
  123         :: a
  124         -- ^ The unique key associated with this entry.
  125     , entryValue
  126         :: Coin
  127         -- ^ The coin value associated with this entry.
  128     } deriving (Eq, Generic, Ord, Show)
  129 
  130 -- | Constructs a 'CoinMap' from a list of entries.
  131 --
  132 -- See 'CoinMapEntry'.
  133 --
  134 coinMapFromList :: Ord a => [CoinMapEntry a] -> CoinMap a
  135 coinMapFromList = CoinMap
  136     . Map.fromListWith (<>)
  137     . fmap (entryKey &&& entryValue)
  138 
  139 -- | Converts a 'CoinMap' to a list of entries.
  140 --
  141 -- See 'CoinMapEntry'.
  142 --
  143 coinMapToList :: CoinMap a -> [CoinMapEntry a]
  144 coinMapToList = fmap (uncurry CoinMapEntry) . Map.toList . unCoinMap
  145 
  146 -- | Calculates the total coin value associated with a 'CoinMap'.
  147 --
  148 coinMapValue :: CoinMap a -> Coin
  149 coinMapValue = mconcat . fmap entryValue . coinMapToList

  151 --------------------------------------------------------------------------------
  152 -- Coin Selection
  153 --------------------------------------------------------------------------------
  154 
  155 -- | Provides a __common interface__ for coin selection algorithms.
  156 --
  157 -- The function 'selectCoins', when applied to the given
  158 -- 'CoinSelectionParameters' object (with /available inputs/ and /requested/
  159 -- /outputs/), will generate a 'CoinSelectionResult' (with /remaining inputs/
  160 -- and a /coin selection/).
  161 --
  162 newtype CoinSelectionAlgorithm i o m = CoinSelectionAlgorithm
  163     { selectCoins
  164         :: CoinSelectionParameters i o
  165         -> ExceptT CoinSelectionError m (CoinSelectionResult i o)
  166     }
  167 
  168 -- | The complete set of parameters required for a 'CoinSelectionAlgorithm'.
  169 --
  170 -- The 'inputsAvailable' and 'outputsRequested' fields are both maps of unique
  171 -- keys to associated 'Coin' values, where:
  172 --
  173 --   * Each key-value pair in the 'inputsAvailable' map corresponds to an
  174 --     __unspent output__ from a previous transaction that is /available/
  175 --     /for selection as an input/ by the coin selection algorithm. The /key/
  176 --     is a unique reference to that output, and the /value/ is the amount of
  177 --     unspent value associated with it.
  178 --
  179 --   * Each key-value pair in the 'outputsRequested' map corresponds to a
  180 --     __payment__ whose value is /to be paid for/ by the coin selection
  181 --     algorithm. The /key/ is a unique reference to a payment recipient,
  182 --     and the /value/ is the amount of money to pay to that recipient.
  183 --
  184 -- A coin selection algorithm will select a __subset__ of inputs from
  185 -- 'inputsAvailable' in order to pay for __all__ the outputs in
  186 -- 'outputsRequested', where:
  187 --
  188 --   * Inputs __selected__ by the algorithm are included in the 'inputs'
  189 --     set of the generated 'CoinSelection'.
  190 --
  191 --   * Inputs __not__ selected by the algorithm are included in the
  192 --     'inputsRemaining' set of the 'CoinSelectionResult'.
  193 --
  194 -- The number of inputs that can selected is limited by 'limit'.
  195 --
  196 -- The total value of 'inputsAvailable' must be /greater than or equal to/
  197 -- the total value of 'outputsRequested', as given by the 'coinMapValue'
  198 -- function.
  199 --
  200 data CoinSelectionParameters i o = CoinSelectionParameters
  201     { inputsAvailable :: CoinMap i
  202         -- ^ The set of inputs available for selection.
  203     , outputsRequested :: CoinMap o
  204         -- ^ The set of outputs requested for payment.
  205     , limit :: CoinSelectionLimit
  206         -- ^ A limit on the number of inputs that can be selected.
  207     }
  208     deriving Generic
  209 
  210 -- | Represents the __result__ of running a coin selection algorithm.
  211 --
  212 -- See 'CoinSelectionAlgorithm'.
  213 --
  214 data CoinSelectionResult i o = CoinSelectionResult
  215     { coinSelection :: CoinSelection i o
  216         -- ^ The generated coin selection.
  217     , inputsRemaining :: CoinMap i
  218         -- ^ The set of inputs that were __not__ selected.
  219     }
  220 
  221 -- | A __coin selection__ is the basis for a /transaction/.
  222 --
  223 -- It consists of a selection of 'inputs', 'outputs', and 'change'.
  224 --
  225 -- The 'inputs' and 'outputs' fields are both maps of unique keys to associated
  226 -- 'Coin' values, where:
  227 --
  228 --   * Each key-value pair in the 'inputs' map corresponds to an
  229 --     __unspent output__ from a previous transaction (also known as a UTxO).
  230 --     The /key/ is a unique reference to that output, and the /value/ is the
  231 --     amount of unspent value associated with it.
  232 --
  233 --   * Each key-value pair in the 'outputs' map corresponds to a __payment__.
  234 --     The /key/ is a unique reference to a payment recipient, and the /value/
  235 --     is the amount of money to pay to that recipient.
  236 --
  237 -- The 'change' field is a set of coins to be returned to the originator of the
  238 -- transaction.
  239 --
  240 -- The 'CoinSelectionAlgorithm' type provides a common interface for generating
  241 -- coin selections.
  242 --
  243 data CoinSelection i o = CoinSelection
  244     { inputs :: CoinMap i
  245       -- ^ The set of inputs.
  246     , outputs :: CoinMap o
  247       -- ^ The set of outputs.
  248     , change :: [Coin]
  249       -- ^ The set of change.
  250     }
  251     deriving (Generic, Show, Eq)
  252 
  253 instance (Ord i, Ord o) => Semigroup (CoinSelection i o) where
  254     a <> b = CoinSelection
  255         { inputs = inputs a <> inputs b
  256         , outputs = outputs a <> outputs b
  257         , change = change a <> change b
  258         }
  259 
  260 instance (Ord i, Ord o) => Monoid (CoinSelection i o) where
  261     mempty = CoinSelection mempty mempty mempty
  262 
  263 -- | Calculate the total sum of all 'inputs' for the given 'CoinSelection'.
  264 sumInputs :: CoinSelection i o -> Coin
  265 sumInputs = coinMapValue . inputs
  266 
  267 -- | Calculate the total sum of all 'outputs' for the given 'CoinSelection'.
  268 sumOutputs :: CoinSelection i o -> Coin
  269 sumOutputs =  coinMapValue . outputs
  270 
  271 -- | Calculate the total sum of all 'change' for the given 'CoinSelection'.
  272 sumChange :: CoinSelection i o -> Coin
  273 sumChange = mconcat . change
  274 
  275 -- | Defines an __inclusive upper bound__ on the /number/ of inputs that
  276 --   a 'CoinSelectionAlgorithm' is allowed to select.
  277 --
  278 newtype CoinSelectionLimit = CoinSelectionLimit
  279     { calculateLimit
  280         :: Word8 -> Word8
  281             -- ^ Calculate the maximum number of inputs allowed for a given
  282             -- number of outputs.
  283     } deriving Generic
  284 
  285 -- | Represents the set of possible failures that can occur when attempting
  286 --   to produce a 'CoinSelection' with a 'CoinSelectionAlgorithm'.
  287 --
  288 -- See 'selectCoins'.
  289 --
  290 data CoinSelectionError
  291     = InputValueInsufficient
  292         InputValueInsufficientError
  293     | InputCountInsufficient
  294         InputCountInsufficientError
  295     | InputLimitExceeded
  296         InputLimitExceededError
  297     | InputsExhausted
  298         InputsExhaustedError
  299     deriving (Eq, Show)
  300 
  301 -- | Indicates that the total value of 'inputsAvailable' is less than the total
  302 --   value of 'outputsRequested', making it /impossible/ to cover all payments,
  303 --   /regardless/ of which algorithm is chosen.
  304 --
  305 data InputValueInsufficientError =
  306     InputValueInsufficientError
  307     { inputValueAvailable :: Coin
  308         -- ^ The total value of 'inputsAvailable'.
  309     , inputValueRequired :: Coin
  310         -- ^ The total value of 'outputsRequested'.
  311     }
  312     deriving (Eq, Show)
  313 
  314 -- | Indicates that the total count of entries in 'inputsAvailable' is /fewer/
  315 --   /than/ required by the algorithm. The number required depends on the
  316 --   particular algorithm implementation.
  317 --
  318 data InputCountInsufficientError =
  319     InputCountInsufficientError
  320     { inputCountAvailable :: Natural
  321         -- ^ The number of entries in 'inputsAvailable'.
  322     , inputCountRequired :: Natural
  323         -- ^ The number of entries required.
  324     }
  325     deriving (Eq, Show)
  326 
  327 -- | Indicates that all available entries in 'inputsAvailable' were depleted
  328 --   /before/ all the payments in 'outputsRequested' could be paid for.
  329 --
  330 -- This condition can occur /even if/ the total value of 'inputsAvailable' is
  331 -- greater than or equal to the total value of 'outputsRequested', due to
  332 -- differences in the way that algorithms select inputs.
  333 --
  334 data InputsExhaustedError =
  335     InputsExhaustedError
  336     deriving (Eq, Show)
  337 
  338 -- | Indicates that the coin selection algorithm is unable to cover the total
  339 --   value of 'outputsRequested' without exceeding the maximum number of inputs
  340 --   defined by 'limit'.
  341 --
  342 -- See 'calculateLimit'.
  343 --
  344 newtype InputLimitExceededError =
  345     InputLimitExceededError
  346     { calculatedInputLimit :: Word8 }
  347     deriving (Eq, Show)

  353 -- Selects an entry at random from a 'CoinMap', returning both the selected
  354 -- entry and the map with the entry removed.
  355 --
  356 -- If the given map is empty, this function returns 'Nothing'.
  357 --
  358 coinMapRandomEntry
  359     :: MonadRandom m
  360     => CoinMap a
  361     -> m (Maybe (CoinMapEntry a), CoinMap a)
  362 coinMapRandomEntry (CoinMap m)
  363     | Map.null m =
  364         return (Nothing, CoinMap m)
  365     | otherwise = do
  366         ix <- fromEnum <$> generateBetween 0 (toEnum (Map.size m - 1))
  367         let entry = uncurry CoinMapEntry $ Map.elemAt ix m
  368         let remainder = CoinMap $ Map.deleteAt ix m
  369         return (Just entry, remainder)
