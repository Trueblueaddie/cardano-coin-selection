   46 -- | An implementation of the __Largest-First__ coin selection algorithm.
   47 --
   48 -- = Overview
   49 --
   50 -- The __Largest-First__ algorithm processes outputs in /descending order of/
   51 -- /value/, from /largest/ to /smallest/.
   52 --
   53 -- For each output, it repeatedly selects the /largest/ remaining unspent UTxO
   54 -- entry until the value of selected entries is greater than or equal to the
   55 -- value of that output.
   56 --
   57 -- = State Maintained by the Algorithm
   58 --
   59 -- At all stages of processing, the algorithm maintains:
   60 --
   61 --  1.  A __/remaining UTxO list/__
   62 --
   63 --      This is initially equal to the given /initial UTxO set/ parameter,
   64 --      sorted into /descending order of coin value/.
   65 --
   66 --      The /head/ of the list is always the remaining UTxO entry with the
   67 --      /largest coin value/.
   68 --
   69 --      Entries are incrementally removed from the /head/ of the list as the
   70 --      algorithm proceeds, until the list is empty.
   71 --
   72 --  2.  An __/unpaid output list/__
   73 --
   74 --      This is initially equal to the given /output list/ parameter, sorted
   75 --      into /descending order of coin value/.
   76 --
   77 --      The /head/ of the list is always the unpaid output with the
   78 --      /largest coin value/.
   79 --
   80 --      Entries are incrementally removed from the /head/ of the list as the
   81 --      algorithm proceeds, until the list is empty.
   82 --
   83 --  3.  An __/accumulated coin selection/__
   84 --
   85 --      This is initially /empty/.
   86 --
   87 --      Entries are incrementally added as each output is paid for, until the
   88 --      /unpaid output list/ is empty.
   89 --
   90 -- = Cardinality Rules
   91 --
   92 -- The algorithm requires that:
   93 --
   94 --  1.  Each output from the given /output list/ is paid for by /one or more/
   95 --      entries from the /initial UTxO set/.
   96 --
   97 --  2.  Each entry from the /initial UTxO set/ is used to pay for /at most one/
   98 --      output from the given /output list/.
   99 --
  100 --      (A single UTxO entry __cannot__ be used to pay for multiple outputs.)
  101 --
  102 -- = Order of Processing
  103 --
  104 -- The algorithm proceeds according to the following sequence of steps:
  105 --
  106 --  *   /Step 1/
  107 --
  108 --      Remove a single /unpaid output/ from the head of the
  109 --      /unpaid output list/.
  110 --
  111 --  *   /Step 2/
  112 --
  113 --      Repeatedly remove UTxO entries from the head of the
  114 --      /remaining UTxO list/ until the total value of entries removed is
  115 --      /greater than or equal to/ the value of the /removed output/.
  116 --
  117 --  *   /Step 3/
  118 --
  119 --      Use the /removed UTxO entries/ to pay for the /removed output/.
  120 --
  121 --      This is achieved by:
  122 --
  123 --      *  adding the /removed UTxO entries/ to the 'inputs' field of the
  124 --         /accumulated coin selection/.
  125 --      *  adding the /removed output/ to the 'outputs' field of the
  126 --         /accumulated coin selection/.
  127 --
  128 --  *   /Step 4/
  129 --
  130 --      If the /total value/ of the /removed UTxO entries/ is greater than the
  131 --      value of the /removed output/, generate a coin whose value is equal to
  132 --      the exact difference, and add it to the 'change' field of the
  133 --      /accumulated coin selection/.
  134 --
  135 --  *   /Step 5/
  136 --
  137 --      If the /unpaid output list/ is empty, __terminate__ here.
  138 --
  139 --      Otherwise, return to /Step 1/.
  140 --
  141 -- = Termination
  142 --
  143 -- The algorithm terminates __successfully__ if the /remaining UTxO list/ is
  144 -- not depleted before the /unpaid output list/ can be fully depleted (i.e., if
  145 -- all the outputs have been paid for).
  146 --
  147 -- The /accumulated coin selection/ and /remaining UTxO list/ are returned to
  148 -- the caller.
  149 --
  150 -- === Failure Modes
  151 --
  152 -- The algorithm terminates with an __error__ if:
  153 --
  154 --  1.  The /total value/ of the initial UTxO set (the amount of money
  155 --      /available/) is /less than/ the total value of the output list (the
  156 --      amount of money /required/).
  157 --
  158 --      See: __'InputValueInsufficientError'__.
  159 --
  160 --  2.  The /number/ of entries in the initial UTxO set is /smaller than/ the
  161 --      number of requested outputs.
  162 --
  163 --      Due to the nature of the algorithm, /at least one/ UTxO entry is
  164 --      required /for each/ output.
  165 --
  166 --      See: __'InputCountInsufficientError'__.
  167 --
  168 --  3.  Due to the particular /distribution/ of values within the initial UTxO
  169 --      set, the algorithm depletes all entries from the UTxO set /before/ it
  170 --      is able to pay for all requested outputs.
  171 --
  172 --      See: __'InputsExhaustedError'__.
  173 --
  174 --  4.  The /number/ of UTxO entries needed to pay for the requested outputs
  175 --      would /exceed/ the upper limit specified by 'limit'.
  176 --
  177 --      See: __'InputLimitExceededError'__.
  178 --
  179 largestFirst
  180     :: (Ord i, Ord o, Monad m)
  181     => CoinSelectionAlgorithm i o m
  182 largestFirst = CoinSelectionAlgorithm payForOutputs
  183 
  184 payForOutputs
  185     :: (Ord i, Ord o, Monad m)
  186     => CoinSelectionParameters i o
  187     -> ExceptT CoinSelectionError m (CoinSelectionResult i o)
  188 payForOutputs params =
  189     case foldM payForOutput (utxoDescending, mempty) outputsDescending of
  190         Just (utxoRemaining, selection) ->
  191             pure $ CoinSelectionResult selection $ coinMapFromList utxoRemaining
  192         Nothing ->
  193             throwE errorCondition
  194   where
  195     errorCondition
  196       | amountAvailable < amountRequested =
  197           InputValueInsufficient $
  198               InputValueInsufficientError
  199                   amountAvailable amountRequested
  200       | utxoCount < outputCount =
  201           InputCountInsufficient $
  202               InputCountInsufficientError
  203                   utxoCount outputCount
  204       | utxoCount <= inputCountMax =
  205           InputsExhausted
  206               InputsExhaustedError
  207       | otherwise =
  208           InputLimitExceeded $
  209               InputLimitExceededError $
  210                   fromIntegral inputCountMax
  211     amountAvailable =
  212         coinMapValue $ inputsAvailable params
  213     amountRequested =
  214         coinMapValue $ outputsRequested params
  215     inputCountMax = fromIntegral
  216         $ calculateLimit (limit params)
  217         $ fromIntegral outputCount
  218     outputCount =
  219         fromIntegral $ length $ coinMapToList $ outputsRequested params
  220     outputsDescending =
  221         L.sortOn (Down . entryValue) $ coinMapToList $ outputsRequested params
  222     utxoCount =
  223         fromIntegral $ L.length $ coinMapToList $ inputsAvailable params
  224     utxoDescending =
  225         take (fromIntegral inputCountMax)
  226             $ L.sortOn (Down . entryValue)
  227             $ coinMapToList
  228             $ inputsAvailable params

  230 -- | Attempts to pay for a /single transaction output/ by selecting the
  231 --   /smallest possible/ number of entries from the /head/ of the given
  232 --   UTxO list.
  233 --
  234 -- Returns a /reduced/ list of UTxO entries, and a coin selection that is
  235 -- /updated/ to include the payment.
  236 --
  237 -- If the total value of entries in the given UTxO list is /less than/ the
  238 -- required output amount, this function will return 'Nothing'.
  239 --
  240 payForOutput
  241     :: forall i o . (Ord i, Ord o)
  242     => ([CoinMapEntry i], CoinSelection i o)
  243     -> CoinMapEntry o
  244     -> Maybe ([CoinMapEntry i], CoinSelection i o)
  245 payForOutput (utxoAvailable, currentSelection) out =
  246     coverTarget utxoAvailable mempty
  247   where
  248     coverTarget
  249         :: [CoinMapEntry i]
  250         -> [CoinMapEntry i]
  251         -> Maybe ([CoinMapEntry i], CoinSelection i o)
  252     coverTarget utxoRemaining utxoSelected
  253         | valueSelected >= valueTarget = Just
  254             -- We've selected enough to cover the target, so stop here.
  255             ( utxoRemaining
  256             , currentSelection <> CoinSelection
  257                 { inputs  = coinMapFromList utxoSelected
  258                 , outputs = coinMapFromList [out]
  259                 , change  = filter (> C.zero)
  260                     (F.toList $ valueSelected `C.sub` valueTarget)
  261                 }
  262             )
  263         | otherwise =
  264             -- We haven't yet selected enough to cover the target, so attempt
  265             -- to select a little more and then continue.
  266             case utxoRemaining of
  267                 utxoEntry : utxoRemaining' ->
  268                     coverTarget utxoRemaining' (utxoEntry : utxoSelected)
  269                 [] ->
  270                     -- The UTxO has been exhausted, so stop here.
  271                     Nothing
  272       where
  273         valueTarget
  274             = entryValue out
  275         valueSelected
  276             = sumEntries utxoSelected
  277 
  278 sumEntries :: [CoinMapEntry a] -> Coin
  279 sumEntries entries = mconcat $ entryValue <$> entries
