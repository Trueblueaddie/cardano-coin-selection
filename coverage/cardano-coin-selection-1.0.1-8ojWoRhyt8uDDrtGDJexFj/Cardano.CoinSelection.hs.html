   91 --------------------------------------------------------------------------------
   92 -- Coin Map
   93 --------------------------------------------------------------------------------
   94 
   95 -- | A mapping from unique keys to associated 'Coin' values.
   96 --
   97 -- A 'CoinMap' can be used to represent:
   98 --
   99 --   * a UTxO set, where each key within the map refers to an unspent output
  100 --     from a previous transaction.
  101 --
  102 --   * a set of 'inputs' to a 'CoinSelection', where each input is an entry
  103 --     selected from a UTxO set by a 'CoinSelectionAlgorithm'.
  104 --
  105 --   * a set of 'outputs' for a 'CoinSelection', where each key within the map
  106 --     refers to the address of a payment recipient.
  107 --
  108 -- A 'CoinMap' can be constructed with the 'coinMapFromList' function.
  109 --
  110 -- The total value of a 'CoinMap' is given by the 'coinMapValue' function.
  111 --
  112 -- @since 1.0.0
  113 newtype CoinMap a = CoinMap { unCoinMap :: Map a Coin }
  114     deriving (Eq, Generic)
  115     deriving Show via (Quiet (CoinMap a))
  116 
  117 instance Foldable CoinMap where
  118     foldMap f = F.fold . fmap (f . entryKey) . coinMapToList
  119 
  120 instance Ord a => Monoid (CoinMap a) where
  121     mempty = CoinMap mempty
  122 
  123 instance Ord a => Semigroup (CoinMap a) where
  124     CoinMap a <> CoinMap b = CoinMap $ Map.unionWith (<>) a b
  125 
  126 -- | An entry for a 'CoinMap'.
  127 --
  128 -- @since 1.0.0
  129 data CoinMapEntry a = CoinMapEntry
  130     { entryKey
  131         :: a
  132         -- ^ The unique key associated with this entry.
  133     , entryValue
  134         :: Coin
  135         -- ^ The coin value associated with this entry.
  136     } deriving (Eq, Generic, Ord, Show)
  137 
  138 -- | Constructs a 'CoinMap' from a list of entries.
  139 --
  140 -- See 'CoinMapEntry'.
  141 --
  142 -- @since 1.0.0
  143 coinMapFromList :: Ord a => [CoinMapEntry a] -> CoinMap a
  144 coinMapFromList = CoinMap
  145     . Map.fromListWith (<>)
  146     . fmap (entryKey &&& entryValue)
  147 
  148 -- | Converts a 'CoinMap' to a list of entries.
  149 --
  150 -- See 'CoinMapEntry'.
  151 --
  152 -- @since 1.0.0
  153 coinMapToList :: CoinMap a -> [CoinMapEntry a]
  154 coinMapToList = fmap (uncurry CoinMapEntry) . Map.toList . unCoinMap
  155 
  156 -- | Calculates the total coin value associated with a 'CoinMap'.
  157 --
  158 -- @since 1.0.0
  159 coinMapValue :: CoinMap a -> Coin
  160 coinMapValue = mconcat . fmap entryValue . coinMapToList

  162 --------------------------------------------------------------------------------
  163 -- Coin Selection
  164 --------------------------------------------------------------------------------
  165 
  166 -- | Provides a __common interface__ for coin selection algorithms.
  167 --
  168 -- The function 'selectCoins', when applied to the given
  169 -- 'CoinSelectionParameters' object (with /available inputs/ and /requested/
  170 -- /outputs/), will generate a 'CoinSelectionResult' (with /remaining inputs/
  171 -- and a /coin selection/).
  172 --
  173 -- For implementations provided by this library, see
  174 -- "Cardano.CoinSelection.Algorithm".
  175 --
  176 -- @since 1.0.0
  177 newtype CoinSelectionAlgorithm i o m = CoinSelectionAlgorithm
  178     { selectCoins
  179         :: CoinSelectionParameters i o
  180         -> ExceptT CoinSelectionError m (CoinSelectionResult i o)
  181     }
  182 
  183 -- | The complete set of parameters required for a 'CoinSelectionAlgorithm'.
  184 --
  185 -- The 'inputsAvailable' and 'outputsRequested' fields are both maps of unique
  186 -- keys to associated 'Coin' values, where:
  187 --
  188 --   * Each key-value pair in the 'inputsAvailable' map corresponds to an
  189 --     __unspent output__ from a previous transaction that is /available/
  190 --     /for selection as an input/ by the coin selection algorithm. The /key/
  191 --     is a unique reference to that output, and the /value/ is the amount of
  192 --     unspent value associated with it.
  193 --
  194 --   * Each key-value pair in the 'outputsRequested' map corresponds to a
  195 --     __payment__ whose value is /to be paid for/ by the coin selection
  196 --     algorithm. The /key/ is a unique reference to a payment recipient,
  197 --     and the /value/ is the amount of money to pay to that recipient.
  198 --
  199 -- A coin selection algorithm will select a __subset__ of inputs from
  200 -- 'inputsAvailable' in order to pay for __all__ the outputs in
  201 -- 'outputsRequested', where:
  202 --
  203 --   * Inputs __selected__ by the algorithm are included in the 'inputs'
  204 --     set of the generated 'CoinSelection'.
  205 --
  206 --   * Inputs __not__ selected by the algorithm are included in the
  207 --     'inputsRemaining' set of the 'CoinSelectionResult'.
  208 --
  209 -- The number of inputs that can selected is limited by 'limit'.
  210 --
  211 -- The total value of 'inputsAvailable' must be /greater than or equal to/
  212 -- the total value of 'outputsRequested', as given by the 'coinMapValue'
  213 -- function.
  214 --
  215 -- @since 1.0.0
  216 data CoinSelectionParameters i o = CoinSelectionParameters
  217     { inputsAvailable :: CoinMap i
  218         -- ^ The set of inputs available for selection.
  219     , outputsRequested :: CoinMap o
  220         -- ^ The set of outputs requested for payment.
  221     , limit :: CoinSelectionLimit
  222         -- ^ A limit on the number of inputs that can be selected.
  223     }
  224     deriving Generic
  225 
  226 -- | Represents the __result__ of running a coin selection algorithm.
  227 --
  228 -- See 'CoinSelectionAlgorithm'.
  229 --
  230 -- @since 1.0.0
  231 data CoinSelectionResult i o = CoinSelectionResult
  232     { coinSelection :: CoinSelection i o
  233         -- ^ The generated coin selection.
  234     , inputsRemaining :: CoinMap i
  235         -- ^ The set of inputs that were __not__ selected.
  236     } deriving (Eq, Show)
  237 
  238 -- | A __coin selection__ is the basis for a /transaction/.
  239 --
  240 -- It consists of a selection of 'inputs', 'outputs', and 'change'.
  241 --
  242 -- The 'inputs' and 'outputs' fields are both maps of unique keys to associated
  243 -- 'Coin' values, where:
  244 --
  245 --   * Each key-value pair in the 'inputs' map corresponds to an
  246 --     __unspent output__ from a previous transaction (also known as a UTxO).
  247 --     The /key/ is a unique reference to that output, and the /value/ is the
  248 --     amount of unspent value associated with it.
  249 --
  250 --   * Each key-value pair in the 'outputs' map corresponds to a __payment__.
  251 --     The /key/ is a unique reference to a payment recipient, and the /value/
  252 --     is the amount of money to pay to that recipient.
  253 --
  254 -- The 'change' field is a set of coins to be returned to the originator of the
  255 -- transaction.
  256 --
  257 -- The 'CoinSelectionAlgorithm' type provides a common interface for generating
  258 -- coin selections.
  259 --
  260 -- @since 1.0.0
  261 data CoinSelection i o = CoinSelection
  262     { inputs :: CoinMap i
  263       -- ^ The set of inputs.
  264     , outputs :: CoinMap o
  265       -- ^ The set of outputs.
  266     , change :: [Coin]
  267       -- ^ The set of change.
  268     }
  269     deriving (Generic, Show, Eq)
  270 
  271 instance (Ord i, Ord o) => Semigroup (CoinSelection i o) where
  272     a <> b = CoinSelection
  273         { inputs = inputs a <> inputs b
  274         , outputs = outputs a <> outputs b
  275         , change = change a <> change b
  276         }
  277 
  278 instance (Ord i, Ord o) => Monoid (CoinSelection i o) where
  279     mempty = CoinSelection mempty mempty mempty
  280 
  281 -- | Calculate the total sum of all 'inputs' for the given 'CoinSelection'.
  282 --
  283 -- @since 1.0.0
  284 sumInputs :: CoinSelection i o -> Coin
  285 sumInputs = coinMapValue . inputs
  286 
  287 -- | Calculate the total sum of all 'outputs' for the given 'CoinSelection'.
  288 --
  289 -- @since 1.0.0
  290 sumOutputs :: CoinSelection i o -> Coin
  291 sumOutputs =  coinMapValue . outputs
  292 
  293 -- | Calculate the total sum of all 'change' for the given 'CoinSelection'.
  294 --
  295 -- @since 1.0.0
  296 sumChange :: CoinSelection i o -> Coin
  297 sumChange = mconcat . change

  299 -- | Defines an __inclusive upper bound__ on the /number/ of inputs that
  300 --   a 'CoinSelectionAlgorithm' is allowed to select.
  301 --
  302 -- @since 1.0.0
  303 newtype CoinSelectionLimit = CoinSelectionLimit
  304     { calculateLimit
  305         :: Word16 -> Word16
  306             -- ^ Calculate the maximum number of inputs allowed for a given
  307             -- number of outputs.
  308     } deriving Generic

  310 -- | Represents the set of possible failures that can occur when attempting
  311 --   to produce a 'CoinSelection' with a 'CoinSelectionAlgorithm'.
  312 --
  313 -- See 'selectCoins'.
  314 --
  315 -- @since 1.0.0
  316 data CoinSelectionError
  317     = InputValueInsufficient
  318         InputValueInsufficientError
  319     | InputCountInsufficient
  320         InputCountInsufficientError
  321     | InputLimitExceeded
  322         InputLimitExceededError
  323     | InputsExhausted
  324         InputsExhaustedError
  325     deriving (Eq, Show)
  326 
  327 -- | Indicates that the total value of 'inputsAvailable' is less than the total
  328 --   value of 'outputsRequested', making it /impossible/ to cover all payments,
  329 --   /regardless/ of which algorithm is chosen.
  330 --
  331 -- @since 1.0.0
  332 data InputValueInsufficientError =
  333     InputValueInsufficientError
  334     { inputValueAvailable :: Coin
  335         -- ^ The total value of 'inputsAvailable'.
  336     , inputValueRequired :: Coin
  337         -- ^ The total value of 'outputsRequested'.
  338     }
  339     deriving (Eq, Show)
  340 
  341 -- | Indicates that the total count of entries in 'inputsAvailable' is /fewer/
  342 --   /than/ required by the algorithm. The number required depends on the
  343 --   particular algorithm implementation.
  344 --
  345 -- @since 1.0.0
  346 data InputCountInsufficientError =
  347     InputCountInsufficientError
  348     { inputCountAvailable :: Natural
  349         -- ^ The number of entries in 'inputsAvailable'.
  350     , inputCountRequired :: Natural
  351         -- ^ The number of entries required.
  352     }
  353     deriving (Eq, Show)
  354 
  355 -- | Indicates that all available entries in 'inputsAvailable' were depleted
  356 --   /before/ all the payments in 'outputsRequested' could be paid for.
  357 --
  358 -- This condition can occur /even if/ the total value of 'inputsAvailable' is
  359 -- greater than or equal to the total value of 'outputsRequested', due to
  360 -- differences in the way that algorithms select inputs.
  361 --
  362 -- @since 1.0.0
  363 data InputsExhaustedError =
  364     InputsExhaustedError
  365     deriving (Eq, Show)
  366 
  367 -- | Indicates that the coin selection algorithm is unable to cover the total
  368 --   value of 'outputsRequested' without exceeding the maximum number of inputs
  369 --   defined by 'limit'.
  370 --
  371 -- See 'calculateLimit'.
  372 --
  373 -- @since 1.0.0
  374 newtype InputLimitExceededError =
  375     InputLimitExceededError
  376     { calculatedInputLimit :: Word16 }
  377     deriving (Eq, Show)

  383 -- Selects an entry at random from a 'CoinMap', returning both the selected
  384 -- entry and the map with the entry removed.
  385 --
  386 -- If the given map is empty, this function returns 'Nothing'.
  387 --
  388 coinMapRandomEntry
  389     :: MonadRandom m
  390     => CoinMap a
  391     -> m (Maybe (CoinMapEntry a, CoinMap a))
  392 coinMapRandomEntry (CoinMap m)
  393     | Map.null m =
  394         return Nothing
  395     | otherwise = Just <$> do
  396         ix <- fromEnum <$> generateBetween 0 (toEnum (Map.size m - 1))
  397         let entry = uncurry CoinMapEntry $ Map.elemAt ix m
  398         let remainder = CoinMap $ Map.deleteAt ix m
  399         return (entry, remainder)
