   47 -- | An implementation of the __Largest-First__ coin selection algorithm.
   48 --
   49 -- The Largest-First coin selection algorithm considers available inputs in
   50 -- /descending/ order of value, from /largest/ to /smallest/.
   51 --
   52 -- When applied to a set of requested outputs, the algorithm repeatedly selects
   53 -- entries from the available inputs set until the total value of selected
   54 -- entries is greater than or equal to the total value of requested outputs.
   55 --
   56 -- === Change Values
   57 --
   58 -- If the total value of selected inputs is /greater than/ the total value of
   59 -- all requested outputs, the 'change' set of the resulting selection will
   60 -- contain /a single coin/ with the excess value.
   61 --
   62 -- If the total value of selected inputs is /exactly equal to/ the total value
   63 -- of all requested outputs, the 'change' set of the resulting selection will
   64 -- be /empty/.
   65 --
   66 -- === Failure Modes
   67 --
   68 -- The algorithm terminates with an __error__ if:
   69 --
   70 --  1.  The /total value/ of 'inputsAvailable' (the amount of money
   71 --      /available/) is /less than/ the total value of 'outputsRequested' (the
   72 --      amount of money /required/).
   73 --
   74 --      See: __'InputValueInsufficientError'__.
   75 --
   76 --  2.  It is not possible to cover the total value of 'outputsRequested'
   77 --      without selecting a number of inputs from 'inputsAvailable' that
   78 --      would exceed the maximum defined by 'limit'.
   79 --
   80 --      See: __'InputLimitExceededError'__.
   81 --
   82 -- @since 1.0.0
   83 largestFirst
   84     :: (Ord i, Monad m)
   85     => CoinSelectionAlgorithm i o m
   86 largestFirst = CoinSelectionAlgorithm payForOutputs
   87 
   88 payForOutputs
   89     :: forall i o m . (Ord i, Monad m)
   90     => CoinSelectionParameters i o
   91     -> ExceptT CoinSelectionError m (CoinSelectionResult i o)
   92 payForOutputs params
   93     | amountAvailable < amountRequired =
   94         throwE
   95             $ InputValueInsufficient
   96             $ InputValueInsufficientError amountAvailable amountRequired
   97     | length inputsSelected > inputCountMax =
   98         throwE
   99             $ InputLimitExceeded
  100             $ InputLimitExceededError
  101             $ fromIntegral inputCountMax
  102     | otherwise =
  103         pure CoinSelectionResult {coinSelection, inputsRemaining}
  104   where
  105     amountAvailable =
  106         coinMapValue $ inputsAvailable params
  107     amountRequired =
  108         coinMapValue $ outputsRequested params
  109     coinSelection = CoinSelection
  110         { inputs =
  111             inputsSelected
  112         , outputs =
  113             outputsRequested params
  114         , change = filter (> C.zero)
  115             $ F.toList
  116             $ coinMapValue inputsSelected `C.sub` amountRequired
  117         }
  118     inputsAvailableDescending :: [CoinMapEntry i]
  119     inputsAvailableDescending = inputsAvailable params
  120         & coinMapToList
  121         & L.sortOn (Down . entryValue)
  122     inputCountMax :: Int
  123     inputCountMax = outputsRequested params
  124         & coinMapToList
  125         & length
  126         & fromIntegral @Int @Word16
  127         & calculateLimit (limit params)
  128         & fromIntegral @Word16 @Int
  129     inputsSelected :: CoinMap i
  130     inputsSelected = inputsAvailableDescending
  131         & fmap entryValue
  132         & scanl1 (<>)
  133         & takeUntil (>= amountRequired)
  134         & zip inputsAvailableDescending
  135         & fmap fst
  136         & coinMapFromList
  137     inputsRemaining :: CoinMap i
  138     inputsRemaining = inputsAvailableDescending
  139         & drop (length inputsSelected)
  140         & coinMapFromList

  146 takeUntil :: (a -> Bool) -> [a] -> [a]
  147 takeUntil p = foldr (\x ys -> x : if p x then [] else ys) []
