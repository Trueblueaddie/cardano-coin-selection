    5 -- Provides internal functions relating to verification of invariants.
    6 --
    7 module Internal.Invariant
    8     ( invariant
    9     ) where
   10 
   11 import Prelude
   12 
   13 -- | Checks whether or not an invariant holds, by applying the given predicate
   14 --   to the given value.
   15 --
   16 -- If the invariant does not hold (indicated by the predicate function
   17 -- returning 'False'), throws an error with the specified message.
   18 --
   19 -- >>> invariant "not empty" [1,2,3] (not . null)
   20 -- [1, 2, 3]
   21 --
   22 -- >>> invariant "not empty" [] (not . null)
   23 -- *** Exception: not empty
   24 invariant
   25     :: String
   26         -- ^ The message
   27     -> a
   28         -- ^ The value to test
   29     -> (a -> Bool)
   30         -- ^ The predicate
   31     -> a
   32 invariant msg a predicate =
   33     if predicate a then a else error msg
