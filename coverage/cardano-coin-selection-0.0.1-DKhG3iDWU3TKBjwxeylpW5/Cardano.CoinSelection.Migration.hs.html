   10 -- This module contains an algorithm to select coins for migration from legacy
   11 -- wallets to newer wallets.
   12 --
   13 -- We want users to be able to migrate their funds from a legacy random wallet
   14 -- to a new sequential wallet. To do this, we have to move funds from a wallet
   15 -- to another by making transactions. Funds are ultimately a sum of many coins
   16 -- (a.k.a UTxOs). In a transaction, we can select a few coins, and send them to
   17 -- addresses, effectively creating new coins / UTxOs doing this.
   18 --
   19 -- There are some limitations regarding the number of coins that can be selected
   20 -- at once in a single transaction (theoretically 255 coins, in practice ~170)
   21 -- because there's a transaction max size (in bytes) enforced by the network.
   22 -- Also, there's a direct relationship between the maximum number of inputs we
   23 -- can select, and the maximum number of outputs we can produce (increasing one
   24 -- will decrease the other, and vice-versa).
   25 --
   26 -- When making a transaction, coins used as inputs for a transaction becomes
   27 -- unavailable for a while, until the transaction is inserted into the ledger
   28 -- and, make some new coins available as change (very much like when paying
   29 -- with bank notes to a shop, if we give a 20 EUR note to pay for 3 EUR, we
   30 -- can't spend the remaining 17 EUR before we have received the change!).
   31 -- So, a wallet with a small number of UTxO will not be able to make many
   32 -- transactions in parallel and will have to make them sequentially, waiting
   33 -- for the previous ones to be inserted before making new ones (we also say
   34 -- that a wallet is not "fragmented enough").
   35

   66 -- | Construct a list of coin selections / transactions to transfer the totality
   67 -- of a user's wallet. The resulting 'CoinSelection' do not contain any
   68 -- 'outputs', but only change coins (so there's no restriction about how
   69 -- addresses are generated).
   70 --
   71 -- It tries to fit as many inputs as possible in a single transaction (fixed by
   72 -- the 'Word8' maximum number of inputs given as argument.
   73 --
   74 -- The fee options are used to balance the coin selections and fix a threshold
   75 -- for dust that is removed from the selections.
   76 depleteUTxO
   77     :: forall i o . (Ord i, Ord o)
   78     => FeeOptions i o
   79         -- ^ Fee computation and threshold definition
   80     -> Word8
   81         -- ^ Maximum number of inputs we can select per transaction
   82     -> CoinMap i
   83         -- ^ UTxO to deplete
   84     -> [CoinSelection i o]
   85 depleteUTxO feeOpts batchSize utxo =
   86     evalState migrate (uncurry CoinMapEntry <$> Map.toList (unCoinMap utxo))
   87   where
   88     migrate :: State [CoinMapEntry i] [CoinSelection i o]
   89     migrate = do
   90         batch <- getNextBatch
   91         if null batch then
   92             pure []
   93         else case adjustForFee (mkCoinSelection batch) of
   94             Nothing -> pure []
   95             Just coinSel -> do
   96                 rest <- migrate
   97                 pure (coinSel:rest)
   98 
   99     -- Construct a provisional 'CoinSelection' from the given selected inputs.
  100     -- Note that the selection may look a bit weird at first sight as it has
  101     -- no outputs (we are paying everything to ourselves!).
  102     mkCoinSelection :: [CoinMapEntry i] -> CoinSelection i o
  103     mkCoinSelection inps = CoinSelection
  104         { inputs = coinMapFromList inps
  105         , outputs = mempty
  106         , change =
  107             let chgs = mapMaybe (noDust . entryValue) inps
  108             in if null chgs then [threshold] else chgs
  109         }
  110       where
  111         threshold = Coin $ unDustThreshold $ dustThreshold feeOpts
  112         noDust :: Coin -> Maybe Coin
  113         noDust c
  114             | c < threshold = Nothing
  115             | otherwise = Just c
  116 
  117     -- | Attempt to balance the coin selection by reducing or increasing the
  118     -- change values based on the computed fees.
  119     adjustForFee :: CoinSelection i o -> Maybe (CoinSelection i o)
  120     adjustForFee !coinSel = case change coinSel of
  121         -- If there's no change, nothing to adjust
  122         [] -> Nothing
  123 
  124         -- No difference between required and computed, we're done
  125         (_ : _) | diff == 0 -> Just coinSel
  126 
  127         -- Otherwise, we have 2 cases:
  128         --
  129         -- 1/ diff < 0
  130         -- We aren't giving enough as fee, so we need to reduce one output.
  131         --
  132         -- 2/ diff > 0
  133         -- We have some surplus so we add it to an arbitrary output
  134         --
  135         -- If both cases we can simply modify one output by adding `diff`, the
  136         -- sign of `diff` making for the right modification.
  137         -- We then recursively call ourselves for this might reduce the number
  138         -- of outputs and change the fee.
  139         (c : cs) -> adjustForFee $ coinSel
  140             { change = modifyFirst (c :| cs) (+ diff) }
  141       where
  142         diff :: Integer
  143         diff = actualFee - requiredFee
  144           where
  145             requiredFee = integer $
  146                 unFee $ estimateFee (feeEstimator feeOpts) coinSel
  147             actualFee
  148                 = integer (unCoin $ sumInputs coinSel)
  149                 - integer (unCoin $ sumChange coinSel)
  150 
  151     -- | Apply the given function to the first coin of the list. If the
  152     -- operation makes the 'Coin' smaller than the dust threshold, the coin is
  153     -- discarded.
  154     modifyFirst :: NonEmpty Coin -> (Integer -> Integer) -> [Coin]
  155     modifyFirst (Coin c :| cs) op
  156         | c' < threshold = cs
  157         | otherwise = (Coin (fromIntegral c')):cs
  158       where
  159         c' :: Integer
  160         c' = op (integer c)
  161 
  162         threshold :: Integer
  163         threshold = integer (unDustThreshold (dustThreshold feeOpts))
  164 
  165     getNextBatch :: State [a] [a]
  166     getNextBatch = do
  167         xs <- get
  168         let (batch, rest) = splitAt (fromIntegral batchSize) xs
  169         put rest
  170         pure batch
  171 
  172 -- | Try to find a fix "ideal" number of input transactions that would generate
  173 -- rather balanced transactions.
  174 idealBatchSize :: CoinSelectionOptions i o e -> Word8
  175 idealBatchSize coinselOpts = fixPoint 1
  176   where
  177     fixPoint :: Word8 -> Word8
  178     fixPoint !n
  179         | maxN n <= n = n
  180         | n == maxBound = n
  181         | otherwise = fixPoint (n + 1)
  182       where
  183         maxN :: Word8 -> Word8
  184         maxN = maximumInputCount coinselOpts
  185 
  186 -- | Safe conversion of an integral type to an integer
  187 integer :: Integral a => a -> Integer
  188 integer = fromIntegral
