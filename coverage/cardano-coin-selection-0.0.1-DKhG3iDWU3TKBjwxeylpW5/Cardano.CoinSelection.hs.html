   65 --------------------------------------------------------------------------------
   66 -- Coin
   67 --------------------------------------------------------------------------------
   68 
   69 -- | A non-negative integer value that represents a number of Lovelace.
   70 --
   71 -- One Ada is equal to 1,000,000 Lovelace.
   72 --
   73 newtype Coin = Coin
   74     { unCoin :: Word64 }
   75     deriving stock (Eq, Generic, Ord)
   76     deriving Show via (Quiet Coin)
   77 
   78 instance Monoid Coin where
   79     mempty = minBound
   80 
   81 instance Semigroup Coin where
   82     Coin a <> Coin b = Coin (a + b)
   83 
   84 instance NFData Coin
   85 
   86 instance Bounded Coin where
   87     minBound =
   88         Coin 0
   89     maxBound =
   90         Coin 45_000_000_000_000_000
   91         -- = 45 billion Ada × 1 million Lovelace/Ada:
   92 
   93 coinIsValid :: Coin -> Bool
   94 coinIsValid c = c >= minBound && c <= maxBound

   96 --------------------------------------------------------------------------------
   97 -- Coin Map
   98 --------------------------------------------------------------------------------
   99 
  100 newtype CoinMap a = CoinMap { unCoinMap :: Map a Coin }
  101     deriving (Eq, Generic)
  102     deriving Show via (Quiet (CoinMap a))
  103 
  104 instance Foldable CoinMap where
  105     foldMap f = F.fold . fmap (f . entryKey) . coinMapToList
  106 
  107 instance Ord a => Monoid (CoinMap a) where
  108     mempty = CoinMap mempty
  109 
  110 instance Ord a => Semigroup (CoinMap a) where
  111     CoinMap a <> CoinMap b = CoinMap $ Map.unionWith (<>) a b
  112 
  113 data CoinMapEntry a = CoinMapEntry
  114     { entryKey
  115         :: a
  116     , entryValue
  117         :: Coin
  118     } deriving (Eq, Generic, Ord, Show)
  119 
  120 coinMapFromList :: Ord a => [CoinMapEntry a] -> CoinMap a
  121 coinMapFromList = CoinMap
  122     . Map.fromListWith (<>)
  123     . fmap (entryKey &&& entryValue)
  124 
  125 coinMapToList :: CoinMap a -> [CoinMapEntry a]
  126 coinMapToList = fmap (uncurry CoinMapEntry) . Map.toList . unCoinMap
  127 
  128 coinMapValue :: CoinMap a -> Coin
  129 coinMapValue = mconcat . fmap entryValue . coinMapToList
  130 
  131 -- | Selects an entry at random from a 'CoinMap', returning both the selected
  132 --   entry and the map with the entry removed.
  133 --
  134 -- If the given map is empty, this function returns 'Nothing'.
  135 --
  136 coinMapRandomEntry
  137     :: MonadRandom m
  138     => CoinMap a
  139     -> m (Maybe (CoinMapEntry a), CoinMap a)
  140 coinMapRandomEntry (CoinMap m)
  141     | Map.null m =
  142         return (Nothing, CoinMap m)
  143     | otherwise = do
  144         ix <- fromEnum <$> generateBetween 0 (toEnum (Map.size m - 1))
  145         let entry = uncurry CoinMapEntry $ Map.elemAt ix m
  146         let remainder = CoinMap $ Map.deleteAt ix m
  147         return (Just entry, remainder)

  150 -- Coin Selection
  151 --------------------------------------------------------------------------------

  153 -- | Represents a /coin selection algorithm/.
  154 --
  155 -- The function 'selectCoins', when applied to the given /output list/ and
  156 -- /initial UTxO set/, generates a 'CoinSelection' that is capable of paying
  157 -- for all of the outputs, and a /remaining UTxO set/ from which all spent
  158 -- values have been removed.
  159 --
  160 newtype CoinSelectionAlgorithm i o m e = CoinSelectionAlgorithm
  161     { selectCoins
  162         :: CoinSelectionOptions i o e
  163         -> CoinMap i
  164         -> CoinMap o
  165         -> ExceptT (CoinSelectionError e) m (CoinSelection i o, CoinMap i)
  166     }
  167 
  168 -- | Represents the result of running a /coin selection algorithm/.
  169 --
  170 -- See 'CoinSelectionAlgorithm'.
  171 --
  172 data CoinSelection i o = CoinSelection
  173     { inputs :: CoinMap i
  174       -- ^ A /subset/ of the original 'UTxO' that was passed to the coin
  175       -- selection algorithm, containing only the entries that were /selected/
  176       -- by the coin selection algorithm.
  177     , outputs :: CoinMap o
  178       -- ^ The original set of output payments passed to the coin selection
  179       -- algorithm, whose total value is covered by the 'inputs'.
  180     , change :: [Coin]
  181       -- ^ A set of change values to be paid back to the originator of the
  182       -- payment.
  183     }
  184     deriving (Generic, Show, Eq)
  185 
  186 instance (Ord i, Ord o) => Semigroup (CoinSelection i o) where
  187     a <> b = CoinSelection
  188         { inputs = inputs a <> inputs b
  189         , outputs = outputs a <> outputs b
  190         , change = change a <> change b
  191         }
  192 
  193 instance (Ord i, Ord o) => Monoid (CoinSelection i o) where
  194     mempty = CoinSelection mempty mempty mempty
  195 
  196 -- | Calculate the total sum of all 'inputs' for the given 'CoinSelection'.
  197 sumInputs :: CoinSelection i o -> Coin
  198 sumInputs = coinMapValue . inputs
  199 
  200 -- | Calculate the total sum of all 'outputs' for the given 'CoinSelection'.
  201 sumOutputs :: CoinSelection i o -> Coin
  202 sumOutputs =  coinMapValue . outputs
  203 
  204 -- | Calculate the total sum of all 'change' for the given 'CoinSelection'.
  205 sumChange :: CoinSelection i o -> Coin
  206 sumChange = mconcat . change
  207 
  208 -- | Represents a set of options to be passed to a coin selection algorithm.
  209 --
  210 data CoinSelectionOptions i o e = CoinSelectionOptions
  211     { maximumInputCount
  212         :: Word8 -> Word8
  213             -- ^ Calculate the maximum number of inputs allowed for a given
  214             -- number of outputs.
  215     , validate
  216         :: CoinSelection i o -> Either e ()
  217             -- ^ Validate the given coin selection, returning a backend-specific
  218             -- error.
  219     } deriving (Generic)
  220 
  221 -- | Represents the set of possible failures that can occur when attempting
  222 --   to produce a 'CoinSelection'.
  223 --
  224 data CoinSelectionError e
  225     = ErrUtxoBalanceInsufficient Word64 Word64
  226     -- ^ The UTxO balance was insufficient to cover the total payment amount.
  227     --
  228     -- Records the /UTxO balance/, as well as the /total value/ of the payment
  229     -- we tried to make.
  230     --
  231     | ErrUtxoNotFragmentedEnough Word64 Word64
  232     -- ^ The UTxO was not fragmented enough to support the required number of
  233     -- transaction outputs.
  234     --
  235     -- Records the /number/ of UTxO entries, as well as the /number/ of the
  236     -- transaction outputs.
  237     --
  238     | ErrUtxoFullyDepleted
  239     -- ^ Due to the particular distribution of values within the UTxO set, all
  240     -- available UTxO entries were depleted before all the requested
  241     -- transaction outputs could be paid for.
  242     --
  243     | ErrMaximumInputCountExceeded Word64
  244     -- ^ The number of UTxO entries needed to cover the requested payment
  245     -- exceeded the upper limit specified by 'maximumInputCount'.
  246     --
  247     -- Records the value of 'maximumInputCount'.
  248     --
  249     | ErrInvalidSelection e
  250     -- ^ The coin selection generated was reported to be invalid by the backend.
  251     --
  252     -- Records the /backend-specific error/ that occurred while attempting to
  253     -- validate the selection.
  254     --
  255     deriving (Show, Eq)
