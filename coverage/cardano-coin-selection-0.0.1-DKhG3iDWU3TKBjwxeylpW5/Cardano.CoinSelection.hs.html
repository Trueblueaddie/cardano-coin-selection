   70 --------------------------------------------------------------------------------
   71 -- Coin
   72 --------------------------------------------------------------------------------
   73 
   74 -- | A non-negative integer value that represents a number of Lovelace.
   75 --
   76 -- One Ada is equal to 1,000,000 Lovelace.
   77 --
   78 newtype Coin = Coin
   79     { unCoin :: Word64 }
   80     deriving stock (Eq, Generic, Ord)
   81     deriving Show via (Quiet Coin)
   82 
   83 instance Monoid Coin where
   84     mempty = minBound
   85 
   86 instance Semigroup Coin where
   87     Coin a <> Coin b = Coin (a + b)
   88 
   89 instance NFData Coin
   90 
   91 instance Bounded Coin where
   92     minBound =
   93         Coin 0
   94     maxBound =
   95         Coin 45_000_000_000_000_000
   96         -- = 45 billion Ada × 1 million Lovelace/Ada:
   97 
   98 coinIsValid :: Coin -> Bool
   99 coinIsValid c = c >= minBound && c <= maxBound

  101 --------------------------------------------------------------------------------
  102 -- Coin Map
  103 --------------------------------------------------------------------------------
  104 
  105 -- | A mapping from unique keys to associated 'Coin' values.
  106 --
  107 -- A 'CoinMap' can be used to represent:
  108 --
  109 --   * a UTxO set, where each key within the map refers to an unspent output
  110 --     from a previous transaction.
  111 --
  112 --   * a set of 'inputs' to a 'CoinSelection', where each input is an entry
  113 --     selected from a UTxO set by a 'CoinSelectionAlgorithm'.
  114 --
  115 --   * a set of 'outputs' for a 'CoinSelection', where each key within the map
  116 --     refers to the address of a payment recipient.
  117 --
  118 -- A 'CoinMap' can be constructed with the 'coinMapFromList' function.
  119 --
  120 -- The total value of a 'CoinMap' is given by the 'coinMapValue' function.
  121 --
  122 newtype CoinMap a = CoinMap { unCoinMap :: Map a Coin }
  123     deriving (Eq, Generic)
  124     deriving Show via (Quiet (CoinMap a))
  125 
  126 instance Foldable CoinMap where
  127     foldMap f = F.fold . fmap (f . entryKey) . coinMapToList
  128 
  129 instance Ord a => Monoid (CoinMap a) where
  130     mempty = CoinMap mempty
  131 
  132 instance Ord a => Semigroup (CoinMap a) where
  133     CoinMap a <> CoinMap b = CoinMap $ Map.unionWith (<>) a b
  134 
  135 -- | An entry for a 'CoinMap'.
  136 --
  137 data CoinMapEntry a = CoinMapEntry
  138     { entryKey
  139         :: a
  140         -- ^ The unique key associated with this entry.
  141     , entryValue
  142         :: Coin
  143         -- ^ The coin value associated with this entry.
  144     } deriving (Eq, Generic, Ord, Show)
  145 
  146 -- | Constructs a 'CoinMap' from a list of entries.
  147 --
  148 -- See 'CoinMapEntry'.
  149 --
  150 coinMapFromList :: Ord a => [CoinMapEntry a] -> CoinMap a
  151 coinMapFromList = CoinMap
  152     . Map.fromListWith (<>)
  153     . fmap (entryKey &&& entryValue)
  154 
  155 -- | Converts a 'CoinMap' to a list of entries.
  156 --
  157 -- See 'CoinMapEntry'.
  158 --
  159 coinMapToList :: CoinMap a -> [CoinMapEntry a]
  160 coinMapToList = fmap (uncurry CoinMapEntry) . Map.toList . unCoinMap
  161 
  162 -- | Calculates the total coin value associated with a 'CoinMap'.
  163 --
  164 coinMapValue :: CoinMap a -> Coin
  165 coinMapValue = mconcat . fmap entryValue . coinMapToList
  166 
  167 -- | Selects an entry at random from a 'CoinMap', returning both the selected
  168 --   entry and the map with the entry removed.
  169 --
  170 -- If the given map is empty, this function returns 'Nothing'.
  171 --
  172 coinMapRandomEntry
  173     :: MonadRandom m
  174     => CoinMap a
  175     -> m (Maybe (CoinMapEntry a), CoinMap a)
  176 coinMapRandomEntry (CoinMap m)
  177     | Map.null m =
  178         return (Nothing, CoinMap m)
  179     | otherwise = do
  180         ix <- fromEnum <$> generateBetween 0 (toEnum (Map.size m - 1))
  181         let entry = uncurry CoinMapEntry $ Map.elemAt ix m
  182         let remainder = CoinMap $ Map.deleteAt ix m
  183         return (Just entry, remainder)

  185 --------------------------------------------------------------------------------
  186 -- Coin Selection
  187 --------------------------------------------------------------------------------
  188 
  189 -- | Provides a common interface for coin selection algorithms.
  190 --
  191 -- The function 'selectCoins', when applied to the given /initial UTxO set/
  192 -- and /output set/, generates a 'CoinSelection' that is capable of paying
  193 -- for all of the outputs, and a /remaining UTxO set/ from which all spent
  194 -- values have been removed.
  195 --
  196 -- Each entry in the /initial UTxO set/ refers to a unique unspent output from
  197 -- a previous transaction, together with its corresponding value. The algorithm
  198 -- will select from among the entries in this set to pay for entries in the
  199 -- output set, placing the selected entries in the 'inputs' field of the
  200 -- resulting 'CoinSelection'.
  201 --
  202 -- Each entry in the /output set/ refers to a unique payment recipient together
  203 -- with the value to pay to that recipient. The 'outputs' field of the
  204 -- resulting 'CoinSelection' will be equal to this set.
  205 --
  206 -- The total value of the initial UTxO set must be /greater than or equal to/
  207 -- the total value of the output set, as given by the 'coinMapValue' function.
  208 --
  209 newtype CoinSelectionAlgorithm i o m e = CoinSelectionAlgorithm
  210     { selectCoins
  211         :: CoinSelectionOptions i o e
  212         -> CoinMap i
  213         -> CoinMap o
  214         -> ExceptT (CoinSelectionError e) m (CoinSelection i o, CoinMap i)
  215     }
  216 
  217 -- | Represents the /result/ of running a coin selection algorithm.
  218 --
  219 -- See 'CoinSelectionAlgorithm'.
  220 --
  221 data CoinSelection i o = CoinSelection
  222     { inputs :: CoinMap i
  223       -- ^ A /subset/ of the original UTxO set that was passed to the coin
  224       -- selection algorithm, containing only the entries that were /selected/
  225       -- by the coin selection algorithm.
  226     , outputs :: CoinMap o
  227       -- ^ The original set of output payments passed to the coin selection
  228       -- algorithm, whose total value is covered by the 'inputs'.
  229     , change :: [Coin]
  230       -- ^ A set of change values to be paid back to the originator of the
  231       -- payment.
  232     }
  233     deriving (Generic, Show, Eq)
  234 
  235 instance (Ord i, Ord o) => Semigroup (CoinSelection i o) where
  236     a <> b = CoinSelection
  237         { inputs = inputs a <> inputs b
  238         , outputs = outputs a <> outputs b
  239         , change = change a <> change b
  240         }
  241 
  242 instance (Ord i, Ord o) => Monoid (CoinSelection i o) where
  243     mempty = CoinSelection mempty mempty mempty
  244 
  245 -- | Calculate the total sum of all 'inputs' for the given 'CoinSelection'.
  246 sumInputs :: CoinSelection i o -> Coin
  247 sumInputs = coinMapValue . inputs
  248 
  249 -- | Calculate the total sum of all 'outputs' for the given 'CoinSelection'.
  250 sumOutputs :: CoinSelection i o -> Coin
  251 sumOutputs =  coinMapValue . outputs
  252 
  253 -- | Calculate the total sum of all 'change' for the given 'CoinSelection'.
  254 sumChange :: CoinSelection i o -> Coin
  255 sumChange = mconcat . change
  256 
  257 -- | Represents a set of options to be passed to a coin selection algorithm.
  258 --
  259 data CoinSelectionOptions i o e = CoinSelectionOptions
  260     { maximumInputCount
  261         :: Word8 -> Word8
  262             -- ^ Calculate the maximum number of inputs allowed for a given
  263             -- number of outputs.
  264     , validate
  265         :: CoinSelection i o -> Either e ()
  266             -- ^ Validate the given coin selection, returning a backend-specific
  267             -- error.
  268     } deriving (Generic)
  269 
  270 -- | Represents the set of possible failures that can occur when attempting
  271 --   to produce a 'CoinSelection'.
  272 --
  273 data CoinSelectionError e
  274     = ErrUtxoBalanceInsufficient Word64 Word64
  275     -- ^ The UTxO balance was insufficient to cover the total payment amount.
  276     --
  277     -- Records the /UTxO balance/, as well as the /total value/ of the payment
  278     -- we tried to make.
  279     --
  280     | ErrUtxoNotFragmentedEnough Word64 Word64
  281     -- ^ The UTxO was not fragmented enough to support the required number of
  282     -- transaction outputs.
  283     --
  284     -- Records the /number/ of UTxO entries, as well as the /number/ of the
  285     -- transaction outputs.
  286     --
  287     | ErrUtxoFullyDepleted
  288     -- ^ Due to the particular distribution of values within the UTxO set, all
  289     -- available UTxO entries were depleted before all the requested
  290     -- transaction outputs could be paid for.
  291     --
  292     | ErrMaximumInputCountExceeded Word64
  293     -- ^ The number of UTxO entries needed to cover the requested payment
  294     -- exceeded the upper limit specified by 'maximumInputCount'.
  295     --
  296     -- Records the value of 'maximumInputCount'.
  297     --
  298     | ErrInvalidSelection e
  299     -- ^ The coin selection generated was reported to be invalid by the backend.
  300     --
  301     -- Records the /backend-specific error/ that occurred while attempting to
  302     -- validate the selection.
  303     --
  304     deriving (Show, Eq)
