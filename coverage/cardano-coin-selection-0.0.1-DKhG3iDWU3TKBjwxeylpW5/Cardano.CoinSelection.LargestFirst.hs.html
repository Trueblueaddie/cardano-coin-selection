   42 -- | An implementation of the __Largest-First__ coin selection algorithm.
   43 --
   44 -- = Overview
   45 --
   46 -- The __Largest-First__ algorithm processes outputs in /descending order of/
   47 -- /value/, from /largest/ to /smallest/.
   48 --
   49 -- For each output, it repeatedly selects the /largest/ remaining unspent UTxO
   50 -- entry until the value of selected entries is greater than or equal to the
   51 -- value of that output.
   52 --
   53 -- = State Maintained by the Algorithm
   54 --
   55 -- At all stages of processing, the algorithm maintains:
   56 --
   57 --  1.  A __/remaining UTxO list/__
   58 --
   59 --      This is initially equal to the given /initial UTxO set/ parameter,
   60 --      sorted into /descending order of coin value/.
   61 --
   62 --      The /head/ of the list is always the remaining UTxO entry with the
   63 --      /largest coin value/.
   64 --
   65 --      Entries are incrementally removed from the /head/ of the list as the
   66 --      algorithm proceeds, until the list is empty.
   67 --
   68 --  2.  An __/unpaid output list/__
   69 --
   70 --      This is initially equal to the given /output list/ parameter, sorted
   71 --      into /descending order of coin value/.
   72 --
   73 --      The /head/ of the list is always the unpaid output with the
   74 --      /largest coin value/.
   75 --
   76 --      Entries are incrementally removed from the /head/ of the list as the
   77 --      algorithm proceeds, until the list is empty.
   78 --
   79 --  3.  An __/accumulated coin selection/__
   80 --
   81 --      This is initially /empty/.
   82 --
   83 --      Entries are incrementally added as each output is paid for, until the
   84 --      /unpaid output list/ is empty.
   85 --
   86 -- = Cardinality Rules
   87 --
   88 -- The algorithm requires that:
   89 --
   90 --  1.  Each output from the given /output list/ is paid for by /one or more/
   91 --      entries from the /initial UTxO set/.
   92 --
   93 --  2.  Each entry from the /initial UTxO set/ is used to pay for /at most one/
   94 --      output from the given /output list/.
   95 --
   96 --      (A single UTxO entry __cannot__ be used to pay for multiple outputs.)
   97 --
   98 -- = Order of Processing
   99 --
  100 -- The algorithm proceeds according to the following sequence of steps:
  101 --
  102 --  *   /Step 1/
  103 --
  104 --      Remove a single /unpaid output/ from the head of the
  105 --      /unpaid output list/.
  106 --
  107 --  *   /Step 2/
  108 --
  109 --      Repeatedly remove UTxO entries from the head of the
  110 --      /remaining UTxO list/ until the total value of entries removed is
  111 --      /greater than or equal to/ the value of the /removed output/.
  112 --
  113 --  *   /Step 3/
  114 --
  115 --      Use the /removed UTxO entries/ to pay for the /removed output/.
  116 --
  117 --      This is achieved by:
  118 --
  119 --      *  adding the /removed UTxO entries/ to the 'inputs' field of the
  120 --         /accumulated coin selection/.
  121 --      *  adding the /removed output/ to the 'outputs' field of the
  122 --         /accumulated coin selection/.
  123 --
  124 --  *   /Step 4/
  125 --
  126 --      If the /total value/ of the /removed UTxO entries/ is greater than the
  127 --      value of the /removed output/, generate a coin whose value is equal to
  128 --      the exact difference, and add it to the 'change' field of the
  129 --      /accumulated coin selection/.
  130 --
  131 --  *   /Step 5/
  132 --
  133 --      If the /unpaid output list/ is empty, __terminate__ here.
  134 --
  135 --      Otherwise, return to /Step 1/.
  136 --
  137 -- = Termination
  138 --
  139 -- The algorithm terminates __successfully__ if the /remaining UTxO list/ is
  140 -- not depleted before the /unpaid output list/ can be fully depleted (i.e., if
  141 -- all the outputs have been paid for).
  142 --
  143 -- The /accumulated coin selection/ and /remaining UTxO list/ are returned to
  144 -- the caller.
  145 --
  146 -- === Failure Modes
  147 --
  148 -- The algorithm terminates with an __error__ if:
  149 --
  150 --  1.  The /total value/ of the initial UTxO set (the amount of money
  151 --      /available/) is /less than/ the total value of the output list (the
  152 --      amount of money /required/).
  153 --
  154 --      See: __'ErrUtxoBalanceInsufficient'__.
  155 --
  156 --  2.  The /number/ of entries in the initial UTxO set is /smaller than/ the
  157 --      number of requested outputs.
  158 --
  159 --      Due to the nature of the algorithm, /at least one/ UTxO entry is
  160 --      required /for each/ output.
  161 --
  162 --      See: __'ErrUtxoNotFragmentedEnough'__.
  163 --
  164 --  3.  Due to the particular /distribution/ of values within the initial UTxO
  165 --      set, the algorithm depletes all entries from the UTxO set /before/ it
  166 --      is able to pay for all requested outputs.
  167 --
  168 --      See: __'ErrUtxoFullyDepleted'__.
  169 --
  170 --  4.  The /number/ of UTxO entries needed to pay for the requested outputs
  171 --      would /exceed/ the upper limit specified by 'maximumInputCount'.
  172 --
  173 --      See: __'ErrMaximumInputCountExceeded'__.
  174 --
  175 largestFirst
  176     :: (Ord i, Ord o, Monad m)
  177     => CoinSelectionAlgorithm i o m e
  178 largestFirst = CoinSelectionAlgorithm payForOutputs
  179 
  180 payForOutputs
  181     :: (Ord i, Ord o, Monad m)
  182     => CoinSelectionOptions i o e
  183     -> CoinMap i
  184     -> CoinMap o
  185     -> ExceptT (CoinSelectionError e) m (CoinSelection i o, CoinMap i)
  186 payForOutputs options utxo outputsRequested =
  187     case foldM payForOutput (utxoDescending, mempty) outputsDescending of
  188         Just (utxoRemaining, selection) ->
  189             validateSelection selection $>
  190                 (selection, coinMapFromList utxoRemaining)
  191         Nothing ->
  192             throwE errorCondition
  193   where
  194     errorCondition
  195       | amountAvailable < amountRequested =
  196           ErrUtxoBalanceInsufficient amountAvailable amountRequested
  197       | utxoCount < outputCount =
  198           ErrUtxoNotFragmentedEnough utxoCount outputCount
  199       | utxoCount <= inputCountMax =
  200           ErrUtxoFullyDepleted
  201       | otherwise =
  202           ErrMaximumInputCountExceeded inputCountMax
  203     amountAvailable =
  204         unCoin $ coinMapValue utxo
  205     amountRequested =
  206         unCoin $ coinMapValue outputsRequested
  207     inputCountMax =
  208         fromIntegral $ maximumInputCount options $ fromIntegral outputCount
  209     outputCount =
  210         fromIntegral $ length $ coinMapToList outputsRequested
  211     outputsDescending =
  212         L.sortOn (Down . entryValue) $ coinMapToList outputsRequested
  213     utxoCount =
  214         fromIntegral $ L.length $ coinMapToList utxo
  215     utxoDescending =
  216         take (fromIntegral inputCountMax)
  217             $ L.sortOn (Down . entryValue)
  218             $ coinMapToList utxo
  219     validateSelection =
  220         except . left ErrInvalidSelection . validate options

  222 -- | Attempts to pay for a /single transaction output/ by selecting the
  223 --   /smallest possible/ number of entries from the /head/ of the given
  224 --   UTxO list.
  225 --
  226 -- Returns a /reduced/ list of UTxO entries, and a coin selection that is
  227 -- /updated/ to include the payment.
  228 --
  229 -- If the total value of entries in the given UTxO list is /less than/ the
  230 -- required output amount, this function will return 'Nothing'.
  231 --
  232 payForOutput
  233     :: forall i o . (Ord i, Ord o)
  234     => ([CoinMapEntry i], CoinSelection i o)
  235     -> CoinMapEntry o
  236     -> Maybe ([CoinMapEntry i], CoinSelection i o)
  237 payForOutput (utxoAvailable, currentSelection) out =
  238     let target = fromIntegral $ unCoin $ entryValue out in
  239     coverTarget target utxoAvailable mempty
  240   where
  241     coverTarget
  242         :: Integer
  243         -> [CoinMapEntry i]
  244         -> [CoinMapEntry i]
  245         -> Maybe ([CoinMapEntry i], CoinSelection i o)
  246     coverTarget target utxoRemaining utxoSelected
  247         | target <= 0 = Just
  248             -- We've selected enough to cover the target, so stop here.
  249             ( utxoRemaining
  250             , currentSelection <> CoinSelection
  251                 { inputs  = coinMapFromList utxoSelected
  252                 , outputs = coinMapFromList [out]
  253                 , change  = [Coin $ fromIntegral $ abs target | target < 0]
  254                 }
  255             )
  256         | otherwise =
  257             -- We haven't yet selected enough to cover the target, so attempt
  258             -- to select a little more and then continue.
  259             case utxoRemaining of
  260                 utxoEntry : utxoRemaining' ->
  261                     let utxoSelected' = utxoEntry : utxoSelected
  262                         target' = target -
  263                             fromIntegral (unCoin $ entryValue utxoEntry)
  264                     in
  265                     coverTarget target' utxoRemaining' utxoSelected'
  266                 [] ->
  267                     -- The UTxO has been exhausted, so stop here.
  268                     Nothing
